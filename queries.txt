// Events Table Creation
CREATE TABLE `events` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `title` varchar(255) NOT NULL,
 `description` text NOT NULL,
 `content` text NOT NULL,
 `event_date` datetime NOT NULL,
 `display_image` varchar(255) DEFAULT NULL,
 `created_date` datetime NOT NULL,
 `status` enum('0','1') NOT NULL DEFAULT '1',
 PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=latin1


// Testimonial Table

CREATE TABLE `testimonials` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `comment` varchar(255) NOT NULL,
 `user_id` int(11) NOT NULL,
 `status` enum('0','1') NOT NULL DEFAULT '0',
 `created_date` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1

// Create Training Table

CREATE TABLE `training` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `title` varchar(255) NOT NULL,
 `content` text NOT NULL,
 `nominate` enum('0','1') NOT NULL,
 `is_upcoming` enum('0','1') NOT NULL,
 `training_date` datetime NOT NULL,
 `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
 `status` enum('0','1') NOT NULL DEFAULT '1',
 PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=latin1


// Create Registrations Table

CREATE TABLE `registration` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `name` varchar(255) NOT NULL,
 `mobile` varchar(20) NOT NULL,
 `email` varchar(255) NOT NULL,
 `address` text NOT NULL,
 `location_id` int(11) NOT NULL,
 `college_id` int(11) NOT NULL,
 `course_id` int(11) NOT NULL,
 `year_of_study` enum('1','2','3','4') NOT NULL,
 `branch` varchar(255) NOT NULL,
 `preferred_location_id` int(11) NOT NULL,
 `laptop_status` enum('1','2','3','4') NOT NULL COMMENT '1-Yes,2-No,3-Can Bring For Training,4-Cant bring laptop for training',
 PRIMARY KEY (`id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1

// Create Location Table

CREATE TABLE `location` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `name` varchar(255) NOT NULL,
 `location_code` varchar(20) NOT NULL,
 PRIMARY KEY (`id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1

// Create College Table

CREATE TABLE `college` ( `id` INT NOT NULL AUTO_INCREMENT , `college_name` VARCHAR(255) NOT NULL , PRIMARY KEY (`id`)) ENGINE = MyISAM

// Create Course Table

CREATE TABLE `course` (
